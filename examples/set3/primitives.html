Boolean


let nameFieldChecked = true; // yes, name field is checked
let gameState = false; // off, game is over
let isGreater = 4 > 1;
let isGreater2 = 4 < 1;
console.log(nameFieldChecked, gameState, isGreater, isGreater2, typeof(isGreater));

BigInt


// the "n" at the end means it's a BigInt

console.log(bigInt, typeof(bigInt));

Number


var n = 123;
{
let n = 12.345;
console.log(n, typeof(n), "inside first block");
}
console.log(n, typeof(n), "outside");
// cannot re-declare with let
var n = "abcd";
console.log("var n = 'abcd'; // declaration");
{
let n = 7.3;
console.log(n, typeof(n), "inside second block");
}
console.log(n, typeof(n), "outside");
