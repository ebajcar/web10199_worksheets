var name = "Ellen";
var a = (name)
  ? "Hello, ${name}!"
  : "the result is ${1 + 2}";
console.log("a: ", a);

var b = (name)
  ? 'Hello, ${name}!'
  : 'the result is ${1 + 2}';
console.log("b: ", b);

var c = (name)
  ? `Hello, ${name}!`
  : `the result is ${1 + 2}`;
console.log("c: ", c);

var cc = (name)
  ? "Hello, " + name + "!"
  : `the result is ${1 + 2}`;
console.log("cc: ", cc);

var d = (name === "John")
  ? `Hello, ${name}!`
  : `the result is ${1 + 2}`;
console.log("d: ", d);
