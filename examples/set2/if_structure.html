if control structure


var x = 7;
if (x == 7) {
  console.log("condition x is true");
} else {
  console.log("condition x is false");
}


var y = 7;
if (y == "7") {
  console.log("condition y is true!!!!");
} else {
  console.log("condition y is false");
}



var z = 7;
if (z === "7") {
  console.log("condition z is true");
} else {
  console.log("condition z is false!!!!");
}



var a = 14;
var b = 7;
if (a > b && a == 42) {
  console.log("condition a is true");
} else {
  console.log("condition a is false");
}



if (a > b || a == 42) {
  console.log("condition b is true");
} else {
  console.log("condition b is false");
}


var c = (a > b || a == 42)
  ? "condition c is true"
  : "condition c is false";
console.log(c);
